## 五、常见问题解决

### 问题1：图片名称正确但仍报错
- **解决**：检查图片文件扩展名是否正确（必须是.png）
- **验证**：在文件管理器中开启"显示文件扩展名"选项确认

### 问题2：部分页面图片显示正常，部分页面不显示
- **解决**：检查相对路径是否正确，建议统一使用绝对路径`/images/xxx.png`
- **示例**：正确`/images/arrow-right.png`，错误`../images/arrow-right.png`

### 问题3：图片显示模糊
- **解决**：按建议尺寸准备图片，避免过度缩放
- **最佳实践**：提供2倍图（命名如arrow-right@2x.png）适配高清屏幕

### 问题4：微信开发者工具显示正常，真机测试不显示
- **解决**：检查图片文件是否已添加到版本控制并提交
- **验证**：通过微信开发者工具的"项目详情"→"本地代码"查看文件是否存在

## 六、图片资源管理规范

### 命名规范
- 全部小写字母，单词间用连字符(-)分隔
- 包含功能描述，避免无意义名称（如icon1.png）
- 状态区分：如需不同状态图片，使用后缀区分（如favorite-active.png）

### 格式要求
- 优先使用PNG格式（支持透明背景）
- 色彩模式：RGB模式（不使用CMYK）
- 分辨率：72dpi（屏幕显示无需高分辨率）

### 版本控制
- 所有图片资源必须纳入Git版本控制
- 提交前添加图片到项目目录，不要使用外部链接
- 重大图片更新时在提交信息中注明

## 二、系统性错误解决方案（500错误专项）

### 根本原因分析
所有图片均返回500错误，表明问题可能出在**后端服务配置**而非单纯文件缺失：

1. **静态资源服务未配置**
   - 后端服务器未正确设置静态资源目录
   - 缺少处理图片请求的中间件

2. **文件系统权限问题**
   - 服务器进程无读取图片目录权限
   - 图片文件所有者与服务运行用户不匹配

3. **路径映射错误**
   - 服务器配置中/images路径映射错误
   - 实际文件存放路径与配置不符

### 后端配置修复方案

#### 方案一：Node.js/Express服务器配置
```javascript
// app.js 或 server.js
const express = require('express');
const app = express();
const path = require('path');

// 正确配置静态资源目录
app.use('/images', express.static(path.join(__dirname, 'images')));

// 验证配置是否生效
app.get('/test-image-config', (req, res) => {
  const testImages = [
    'default-avatar.png',
    'arrow-right.png',
    'wallet.png'
  ];
  
  const result = {
    config: 'images mapped to ' + path.join(__dirname, 'images'),
    availableImages: [],
    missingImages: []
  };
  
  testImages.forEach(img => {
    const imgPath = path.join(__dirname, 'images', img);
    if (fs.existsSync(imgPath)) {
      result.availableImages.push(img);
    } else {
      result.missingImages.push(img);
    }
  });
  
  res.json(result);
});
```

#### 方案二：Nginx服务器配置
```nginx
# nginx.conf 或站点配置文件
server {
    listen 80;
    server_name yourdomain.com;
    
    # 静态资源配置
    location /images/ {
        alias /path/to/your/project/images/;  # 绝对路径
        expires 1d;  # 缓存控制
        add_header Cache-Control "public, max-age=86400";
        
        # 日志记录，用于调试
        access_log /var/log/nginx/images-access.log;
        error_log /var/log/nginx/images-error.log;
        
        # 处理500错误的关键配置
        try_files $uri $uri/ =404;  # 避免将不存在的文件转发到后端
    }
}
```

```python
    # 创建占位图片
    im = Image.new('RGBA', img['size'], (245, 247, 250, 255))
    draw = ImageDraw.Draw(im)
    
    # 添加文字提示
    try:
        font = ImageFont.truetype('arial.ttf', size=int(min(img['size'])*0.5))
    except:
        font = ImageFont.load_default()
        
    text = img['text']
    text_width, text_height = draw.textsize(text, font=font)
    text_x = (img['size'][0] - text_width) // 2
    text_y = (img['size'][1] - text_height) // 2
    
    draw.text((text_x, text_y), text, font=font, fill=(153, 153, 153, 255))
    
    # 保存图片
    im.save(image_path)
    print(f"生成占位图片: {image_path}")

print("批量生成完成，共处理图片:", len(image_list))
```

## 四、500错误快速诊断工具

创建`check_image_server.js`文件：
```javascript
// 检查图片服务配置是否正确
const fs = require('fs');
const path = require('path');
const http = require('http');

// 项目根目录
const PROJECT_ROOT = __dirname;
// 图片目录
const IMAGES_DIR = path.join(PROJECT_ROOT, 'images');

// 测试图片列表
const TEST_IMAGES = [
  'default-avatar.png',
  'arrow-right.png',
  'wallet.png'
];

// 检查本地文件
function checkLocalFiles() {
  console.log('====== 本地文件检查 ======');
  const result = {
    total: TEST_IMAGES.length,
    found: 0,
    missing: []
  };
  
  TEST_IMAGES.forEach(img => {
    const imgPath = path.join(IMAGES_DIR, img);
    if (fs.existsSync(imgPath)) {
      console.log(`✅ 找到: ${img}`);
      result.found++;
    } else {
      console.log(`❌ 缺失: ${img}`);
      result.missing.push(img);
    }
  });
  
  console.log(`\n本地文件检查结果: 找到${result.found}/${result.total}个图片\n`);
  return result;
}

// 检查服务器响应
function checkServerResponse() {
  return new Promise((resolve) => {
    console.log('====== 服务器响应检查 ======');
    // 默认开发服务器地址
    const SERVER_URL = 'http://localhost:3000'; // 根据实际情况修改
    const result = {
      total: TEST_IMAGES.length,
      success: 0,
      errors: []
    };
    
    let checked = 0;
    
    TEST_IMAGES.forEach(img => {
      const imgUrl = `${SERVER_URL}/images/${img}`;
      
      http.get(imgUrl, (res) => {
        checked++;
        
        if (res.statusCode === 200) {
          console.log(`✅ ${imgUrl} (${res.statusCode})`);
          result.success++;
        } else {
          console.log(`❌ ${imgUrl} (${res.statusCode})`);
          result.errors.push({
            url: imgUrl,
            statusCode: res.statusCode
          });
        }
        
        if (checked === TEST_IMAGES.length) {
          console.log(`\n服务器响应检查结果: ${result.success}/${result.total}个图片可访问\n`);
          resolve(result);
        }
      }).on('error', (e) => {
        checked++;
        console.log(`❌ ${imgUrl} (连接错误)`);
        result.errors.push({
          url: imgUrl,
          error: e.message
        });
        
        if (checked === TEST_IMAGES.length) {
          resolve(result);
        }
      });
    });
  });
}

// 主函数
async function main() {
  console.log('=== 图片服务诊断工具 ===\n');
  
  // 1. 检查本地文件
  const fileCheckResult = checkLocalFiles();
  if (fileCheckResult.missing.length > 0) {
    console.log('建议: 先补充缺失的图片文件\n');
  }
  
  // 2. 检查服务器响应
  const serverCheckResult = await checkServerResponse();
  
  // 3. 生成诊断报告
  console.log('=== 诊断报告 ===');
  if (serverCheckResult.errors.length === 0) {
    console.log('✅ 图片服务配置正常');
  } else if (serverCheckResult.errors.every(e => e.statusCode === 404)) {
    console.log('❌ 服务器未找到图片资源，可能是:');
    console.log('   1. 静态资源服务未配置');
    console.log('   2. 路径映射错误');
  } else if (serverCheckResult.errors.every(e => e.statusCode === 500)) {
    console.log('❌ 服务器内部错误，可能是:');
    console.log('   1. 服务器权限不足');
    console.log('   2. 后端代码处理图片请求时出错');
    console.log('   3. 中间件配置错误');
  }
  
  console.log('\n诊断完成');
}

// 执行诊断
main();
```
